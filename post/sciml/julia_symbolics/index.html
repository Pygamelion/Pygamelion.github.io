## 定义符号变量
@variables r θ φ

# Cartesian → Spherical 变换
X = [
    r * sin(θ) * cos(φ),
    r * sin(θ) * sin(φ),
    r * cos(θ)
]

Q = [r, θ, φ]

# 欧氏度规 g_ij = δ_ij
g = I(3)     # 3×3 单位矩阵

# 球坐标度规 h = Mᵀ * g * M

# 坐标变量
@variables t x y z
X = [t, x, y, z]

# 度规张量及其逆
g = I(4) .* [-z^(-1 // 2), z, z, z^(-1 // 2)]
gi = inv(g)

# 根据度规计算克氏符
Γ = sum([
    (1 // 2) * gi[σ, ρ] *
    (
        derivative(g[υ, ρ], X[μ]) +
        derivative(g[μ, ρ], X[υ]) -
        derivative(g[μ, υ], X[ρ])
    )  # 公式部分
    for μ in 1:4, υ in 1:4, σ in 1:4]   # 三重数组生成指标
        for ρ in 1:4) .|> simplify  # 缩并求和指标

# 显示所有分量（剔除了0分量和对称分量）
[
    Symbolics.Variable(Symbol("Γ^{$σ}_{$μ$υ}")) ~ Γ[μ, υ, σ]
    for μ in 1:4, υ in 1:4, σ in 1:4
    if !iszero(Γ[μ, υ, σ]) && μ <= υ
] |> latexify|> show

R = [
    derivative(Γ[μ, σ, ρ], X[υ]) -
    derivative(Γ[υ, σ, ρ], X[μ]) +
    sum(
        Γ[μ, σ, λ] * Γ[υ, λ, ρ] -
        Γ[υ, σ, λ] * Γ[μ, λ, ρ]  # 公式部分
        for λ in 1:4
    ) # 第3、4项的缩并求和指标
    for μ in 1:4, υ in 1:4, σ in 1:4, ρ in 1:4] # 四重数组生成指标

# 显示所有分量（剔除了０分量和反对称分量）
sqrt_rule = @rule sqrt((~x)^2) => ~x
@chain begin
    [
        Symbolics.Variable(Symbol("R_{$μ$υ$σ}^{$ρ}")) ~ R[μ, υ, σ, ρ]
        for μ in 1:4, υ in 1:4, σ in 1:4, ρ in 1:4
        if !iszero(R[μ, υ, σ, ρ]) && μ <= υ
    ]

    ### 简化子表达式的方法，翻翻 julia discourse 有差不多的问题
    simplify.(_; rewriter=Rewriters.Prewalk(Rewriters.PassThrough(sqrt_rule)))
    latexify
    show
end

# 待求的Killing矢量场的分量, 使用@syms显示声明它是θ和φ的函数
ξ = @syms ξ₁(θ, φ) ξ₂(θ, φ)

# 坐标变换
X = [sin(θ) * cos(φ), sin(θ) * sin(φ), cos(θ)]
Q = [θ, φ]

# 三维欧氏度规矩阵
g = I(3)

# 计算二维球面上的诱导度规矩阵h，及其逆hi
M = Symbolics.jacobian(X, Q)
h = M' * g * M
hi = inv(h)

# 根据度规计算克氏符
Γ = @chain begin
    sum(
        [
            (1 // 2) * hi[σ, ρ] * (
                derivative(h[υ, ρ], Q[μ]) +
                derivative(h[μ, ρ], Q[υ]) -
                derivative(h[μ, υ], Q[ρ])
            )
            for μ in 1:2, υ in 1:2, σ in 1:2
        ]
        for ρ in 1:2)
    @. substitute(_, cos(φ)^2 => 1 - sin(φ)^2)
    @. expand
    @. simplify
end

# 列出所有独立Killing方程
eqs = @chain begin
    [
        0 ~
            derivative(sum(h[υ, ρ] * ξ[ρ](θ, φ) for ρ in 1:2), Q[μ]) +
            derivative(sum(h[μ, ρ] * ξ[ρ](θ, φ) for ρ in 1:2), Q[υ]) -
            2 * sum(Γ[μ, υ, σ] * sum(h[σ, ρ] * ξ[ρ](θ, φ) for ρ in 1:2)

end

#  三指标Levi-Civita记号
ε₃ = [
    (length(Set([i, j, k])) == 3) *       # 存在重复指标，取0
    (-1)^(sum([i, j, k][σ] > [i, j, k][ρ]      # 计算逆序数，偶数取1，奇数取-1
              for σ in 1:3, ρ in 1:3 if σ < ρ) % 2)
    for i in 1:3, j in 1:3, k in 1:3
]

#  四维Levi-Civita记号
ε₄ = [
    (length(Set([i, j, k, l])) == 4) *       #  存在重复指标，取0
    (-1)^(sum([i, j, k, l][σ] > [i, j, k, l][ρ]   # 计算逆序数，偶数取1，奇数取-1
              for σ in 1:4, ρ in 1:4 if σ < ρ) % 2)
    for i in 1:4, j in 1:4, k in 1:4, l in 1:4
]

### 直接用这个来算嘛...
@inline function levi_civita_tuple(tp::NTuple{N,Int}) where N
    # 重复检查
    @inbounds for i in 1:N-1
        ai = tp[i]
        for j in i+1:N
            if ai == tp[j]
                return Int8(0)
            end
        end
    end

    # 逆序数
    inv = 0
    @inbounds for i in 1:N-1
        ai = tp[i]
        for j in i+1:N
            if ai > tp[j]
                inv += 1
            end
        end
    end

    return (inv & 1) == 0 ? Int8(1) : Int8(-1)
end
